n_jobs = 14
grid_size = 100
beta,alpha,delta = 0.9,0.3,0.05
P = np_array([[0.6, 0.4],[0.3, 0.7]])
zeta_grid = np_array([1, 1.5])

r = lambda x, u, zeta: np_log(u)
T = lambda x, zeta: zeta * x**alpha + (1-delta)*x

def g (x, u, zeta):
    return zeta * (x) ** alpha + (1-delta) * x - u 

def Expected_v(x, u, zeta, x_grid, v0_grid):
    return np.dot(P[0 if zeta == zeta_grid[0] else 1, :], np_array([interp(g(x, u, zeta), x_grid, v) for v in v0_grid]))
    

def value_func_iterate(x_grid, zeta_grid, v0_grid):
    res = np_array(
        Parallel(n_jobs=n_jobs, verbose=0, return_as='list', batch_size=grid_size // n_jobs)
            (delayed(fminbound)
            (func=lambda u, x=x, zeta=zeta: -(r(x, u, zeta) + beta * Expected_v(x, u, zeta, x_grid, v0_grid)), x1=0, x2=T(x,zeta), full_output=True) 
            for zeta in zeta_grid for x in x_grid))
    u_grid, v_grid_updated = np_array(res[:,0]), -np_array(res[:,1])
    u_grid = u_grid.reshape(-1, grid_size)
    v_grid_updated = v_grid_updated.reshape(-1, grid_size)
    return u_grid, v_grid_updated

k_L = (zeta_grid[0]* alpha/(1/beta-1+delta)) ** (1 / (1 - alpha))
k_H = (zeta_grid[1]* alpha/(1/beta-1+delta)) ** (1 / (1 - alpha))

current_k_grid = np.linspace(0.000001, 1.5 * k_H, grid_size)
next_k_grid = np.tile(current_k_grid, (2, 1))
current_y_grid = (np.tile(zeta_grid, (grid_size, 1)).T)*(np.tile(current_k_grid, (2, 1))**alpha)
current_c_grid = current_y_grid - delta * next_k_grid
v_grid = np_log(current_c_grid) / (1 - beta)

max_iter = 200
tol = 10 ** -6
for _ in tqdm.tqdm(range(max_iter), ncols=50):
    u_grid, v_grid = value_func_iterate(x_grid=current_k_grid, zeta_grid=zeta_grid, v0_grid=v_grid)
    g_ = (current_y_grid - u_grid + (1-delta)*np.tile(current_k_grid, (2,1)))
    err = np.max(np.abs(g_ - next_k_grid))
    next_k_grid = g_
    if err < tol:
        break

def next_k(init_k, init_z, k_prime, k):
    return interp(init_k, k, k_prime[0 if init_z == zeta_grid[0] else 1])

def next_zeta(init_zeta):
    return np.random.choice(zeta_grid,1,p= P[0 if init_zeta == zeta_grid[0] else 1, :])

def simulate(k, k_prime, turn, k_list, z_list): 
    if turn == 0:
        return np_array(k_list)[100:]
    k_list.append(next_k(k_list[-1], z_list[-1], k_prime, k))
    z_list.append(next_zeta(z_list[-1]))
    return simulate(k, k_prime, turn-1, k_list, z_list)

k_dist = simulate(current_k_grid, next_k_grid, 1000, [k_L/2], [zeta_grid[0]])

def plot(k_d):
    fig, ax1 = plt.subplots()
    sns.histplot(k_d, ax=ax1, color='lightblue')
    ax2 = ax1.twinx()
    sns.lineplot(x=current_k_grid, y=next_k_grid[0], ax=ax2)
    sns.lineplot(x=current_k_grid, y=next_k_grid[1], ax=ax2)
    sns.lineplot(x=current_k_grid, y=current_k_grid, ax=ax2)
    max_bar_height = current_k_grid.max()
    ax2.set_ylim(0, max_bar_height) 
    ax2.set_xlabel('K')
    ax1.set_xlabel('K')
    ax1.set_ylabel("K'")
    plt.axvline(k_L, 0,9999)
    plt.axvline(k_H, 0,9999)
    plt.show()
