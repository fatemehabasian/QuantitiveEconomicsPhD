simulation_rounds = 100
hyper_simulation_rounds = 1000
s = np.random.uniform(low=0, high=1, size=(hyper_simulation_rounds, simulation_rounds, N)) ** 2
s[:, :, 1] = 3 * s[:, :, 1]
s[:, :, 2] = 5 * s[:, :, 2]
s = s.sum(axis=2) ** 0.5
s = s.mean(axis=1)
sns.displot(x=s, kind="kde", color='r')
plt.vlines(x=s.mean(), ymin=0, ymax=3)
plt.legend(["hyper-simulation", f"Average of all simulations is {s.mean()}"]);
plt.show()

With 100 samples, the precision is about 0.2
